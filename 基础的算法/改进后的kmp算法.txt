#include<iostream>
#include<string>
using namespace std;
void getnext(string T,int *next)
{
	next[1] = 0;
	int j = 0;
	int i = 1;
	while (i < T[0])
	{
		if (j==0||T[i] == T[j])
		{
			i++;
			j++;
			if (T[i] != T[j])
			{
				next[i] = j;
			}
			else {
				next[i] = next[j];
			}
			
		}
		else {
			j = next[j];
		}
	}
	
	
}

int kmp(string S, string T, int pos)   //返回子串T在主串S第pos个字符之后的位置
{
	int i = pos;
	int j = 1;
	int next[255];
	getnext(T, next);
	while (i <= S[0] && j <= T[0])
	{
		if (j==0||S[i] == T[i])
		{
			i++;
			j++;

		}
		else
		{
			j = next[j];
		}
	}
	if (j < T[0])
	{
		return i - T[0];
	}
	else {
		return 0;
	}
}
